SELECT matnr mtart FROM mara UP TO 1 ROWS INTO TABLE lt_table. " ---- INTERNAL TABLE OPERATIONS ---- APPEND ls_line TO lt_table. READ TABLE lt_table INTO ls_line WITH KEY matnr = '100'. LOOP AT lt_table INTO ls_line. " ... process ENDLOOP. SORT lt_table BY matnr ASCENDING. DELETE lt_table WHERE matnr = ''. CLEAR ls_line. REFRESH lt_table. FREE lt_table.
